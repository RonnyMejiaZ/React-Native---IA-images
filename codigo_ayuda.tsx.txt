// @ts-ignore

import {
  ActivityIndicator,
  Dimensions,
  FlatList,
  Image,
  View,
} from "react-native";
import { useEffect, useState } from "react";

const ACCESS_KEY = "BhDen-UgVPwBYaFKw_FFIVnmxKRxEutjJMQrTKhBr_Y";

export type UnsplashImage = {
  id: string;
  urls: {
    small: string;
  };
  user: {
    name: string;
  };
  height: number;
  width: number;
};
const imageWidth = Dimensions.get("window").width / 2 - 20;

export default function GameSlug() {
  const [images, setImages] = useState<UnsplashImage[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch(`https://api.unsplash.com/photos?client_id=${ACCESS_KEY}`)
      .then((res) => res.json())
      .then((data) => {
        setImages(data);
        setLoading(false);
      });
  }, []);

  

  return loading ? (
    <ActivityIndicator color="#000" size={"large"} />
  ) : (
    
    <FlatList
      data={images}
      keyExtractor={(item) => item.id}
      numColumns={2}
      contentContainerStyle={{ }}
      renderItem={({ item, index }) => (
        <View
          style={{
            flex: 1,
            alignItems: "center",
            justifyContent: "center",
          }}
        >
          <Image
            source={{ uri: item.urls.small }}
            style={{
              width: imageWidth,
              height: (item.height / item.width) * imageWidth,
              borderRadius: 10,
              margin: 10,
            }}
            resizeMode="cover"
          />
        </View>
      )}
    />
  );
}
